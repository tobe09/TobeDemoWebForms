CREATE SCHEMA AUTHORIZATION tobedemo
CREATE TABLE tobedemo.users
(
  ROLE_ID                      NVARCHAR2(8)      NOT NULL,
  USERNAME                NVARCHAR2(32)         NOT NULL,
  PASSWORD                NVARCHAR2(32)         NOT NULL,
  FIRST_NAME              NVARCHAR2(32)         NOT NULL,
  LAST_NAME                 NVARCHAR2(32)         NOT NULL,
         );
		 
CREATE TABLE tobedemo.roles
(
ROLE_ID			NVARCHAR2 (8)			NOT NULL,
DESCRIPTION 	NVARCHAR2(32)	NOT NULL
)

CREATE TABLE tobedemo.students
(
   USERNAME      NVARCHAR2 (32) NOT NULL,
   PASSWORD      NVARCHAR2 (32) NOT NULL,
   FIRST_NAME    NVARCHAR2 (32) NOT NULL,
   SURNAME       NVARCHAR2 (32) NOT NULL,
   OTHER_NAME    NVARCHAR2 (32),

   FACULTY       NVARCHAR2 (64) NOT NULL,
   DEPT          NVARCHAR2 (64) NOT NULL,
   DOB           NVARCHAR2 (16) NOT NULL,
   GENDER        NVARCHAR2 (8) NOT NULL,
   MAR_STATUS    NVARCHAR2 (16) NOT NULL,
   PHONE_NO      NVARCHAR2 (16) NOT NULL,
   EMAIL         NVARCHAR2 (64) NOT NULL,
   COUNTRY_ORI   NVARCHAR2 (64) NOT NULL,
   STATE_ORI     NVARCHAR2 (64) NOT NULL,
   STATE_RES     NVARCHAR2 (64) NOT NULL,
   LGA_RES       NVARCHAR2 (64) NOT NULL,
   PRI_SCHOOL    NVARCHAR2 (128) NOT NULL,
   PRI_CERT      NVARCHAR2 (8) NOT NULL,
   SEC_SCHOOL    NVARCHAR2 (128) NOT NULL,
   SEC_CERT      NVARCHAR2 (16) NOT NULL,
   OTHER_INST    NVARCHAR2 (128),
   OTHER_CERT   NVARCHAR2 (16),
   BLOOD_GRP     NVARCHAR2 (8),
   BLOOD_TYPE    NVARCHAR2 (8),
   MED_HIST      NVARCHAR2 (512),
   OTHER_INFO    NVARCHAR2 (512)
);
